Integration

Drop-in for login/_auth-branding.less. The Lottie path ships the runtime once and swaps a .json per customer; the CSS/SVG path is zero-dependency. Both drive the same caption + underline and degrade to a fully composed static state.

▸2 · Lottie init recommended 
// lottie-web is MIT. Ship once, then swap northlake-login.json per customer.
import lottie from 'lottie-web';

const fig  = document.querySelector('[data-nl-entrance]');
const host = fig.querySelector('.auth-branding__mark');
const reduce = matchMedia('(prefers-reduced-motion: reduce)').matches;

fetch('/branding/northlake-login.json')
  .then(r => r.json())
  .then(data => {
    const anim = lottie.loadAnimation({
      container: host, renderer: 'svg',
      loop: false, autoplay: false, animationData: data
    });
    anim.addEventListener('DOMLoaded', () => {
      fig.classList.add('is-playing');            // drives caption + underline
      reduce ? anim.goToAndStop(anim.totalFrames - 1, true)
             : anim.play();
    });
  });
▸3 · Caption CSS + reduced motion .less / .css 
/* The Lottie covers the MARK. The caption + underline are CSS so they
   restyle with theme vars and stay accessible. Animation is additive:
   the composed state is the DEFAULT, so no-JS / reduced-motion = final. */
.nl-wordmark { font-family:'Spectral',serif; font-size:1.95rem; color:#fff; }
.nl-eyebrow  { font-size:var(--font-size-sm); letter-spacing:.36em;
               text-transform:uppercase; color:var(--color-accent); }
.nl-underline{ width:104px; height:2px; background:var(--color-accent);
               transform-origin:left; border-radius:2px; }

.is-playing .nl-wordmark { animation:nl-rise .6s var(--ease-out) 1s both; }
.is-playing .nl-eyebrow  { animation:nl-fade .55s ease 1.26s both; }
.is-playing .nl-underline{ animation:nl-draw .55s var(--ease-out) 1.46s both; }
@keyframes nl-rise { from{opacity:0;transform:translateY(10px)} to{opacity:1;transform:none} }
@keyframes nl-draw { from{transform:scaleX(0)} to{transform:scaleX(1)} }

@media (prefers-reduced-motion: reduce){
  .auth-branding__figure * { animation:none !important; }  /* show composed state */
}
